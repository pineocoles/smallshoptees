BACKEND SETUP

Follow these four steps to connect Supabase. Once done, paste your keys into /assets/supabase.js, commit to GitHub, then return here to sign in.

01

CREATE FREE SUPABASE PROJECT

Go to supabase.com → Sign up → New project → name it smallshoptees. Wait ~2 min for it to spin up.

02

COPY YOUR API KEYS

In Supabase: Project Settings → API. Copy Project URL and anon / public key. Paste both into /assets/supabase.js, then commit & push to GitHub.

03

RUN THE SQL BELOW

In Supabase: SQL Editor → New query. Paste the SQL block below and click Run. Creates all tables and security rules in one shot.

04

INVITE YOUR ADMIN USER

In Supabase: Authentication → Users → Invite user. Enter pineocoles@gmail.com. Check your email, set a password, then come back to /admin/ and sign in.

-- Run ONCE in Supabase SQL Editor → New Query

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ── REVIEWS ──────────────────────────────────────
CREATE TABLE IF NOT EXISTS reviews (
  id            uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  reviewer_name text NOT NULL,
  business      text NOT NULL,
  rating        int  NOT NULL CHECK (rating BETWEEN 1 AND 5),
  body          text NOT NULL,
  visible       boolean NOT NULL DEFAULT true,
  display_order int NOT NULL DEFAULT 0,
  created_at    timestamptz NOT NULL DEFAULT now()
);
ALTER TABLE reviews ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read visible reviews"
  ON reviews FOR SELECT USING (visible = true);
CREATE POLICY "auth manage reviews"
  ON reviews FOR ALL USING (auth.role() = 'authenticated');

-- ── SETTINGS ─────────────────────────────────────
CREATE TABLE IF NOT EXISTS settings (
  key   text PRIMARY KEY,
  value text NOT NULL
);
ALTER TABLE settings ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read settings"
  ON settings FOR SELECT USING (true);
CREATE POLICY "auth manage settings"
  ON settings FOR ALL USING (auth.role() = 'authenticated');
INSERT INTO settings (key, value) VALUES
  ('price_per_shirt', '16'),
  ('min_order_qty',   '50')
ON CONFLICT (key) DO NOTHING;

-- ── CLIENT PROFILES ──────────────────────────────
CREATE TABLE IF NOT EXISTS client_profiles (
  id              uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id         uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  business_name   text NOT NULL,
  contact_name    text NOT NULL,
  email           text NOT NULL,
  phone           text,
  shirt_style     text NOT NULL DEFAULT 'Standard Tee',
  shirt_color     text NOT NULL DEFAULT 'Black',
  total_ordered   int  NOT NULL DEFAULT 0,
  current_qty     int  NOT NULL DEFAULT 0,
  last_order_date date,
  created_at      timestamptz NOT NULL DEFAULT now()
);
ALTER TABLE client_profiles ENABLE ROW LEVEL SECURITY;
CREATE POLICY "user read own profile"
  ON client_profiles FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "user insert own profile"
  ON client_profiles FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "user update own profile"
  ON client_profiles FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "auth read all profiles"
  ON client_profiles FOR SELECT USING (auth.role() = 'authenticated');
CREATE POLICY "auth update all profiles"
  ON client_profiles FOR UPDATE USING (auth.role() = 'authenticated');

-- ── ORDERS ───────────────────────────────────────
CREATE TABLE IF NOT EXISTS orders (
  id             uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id        uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  xs             int  NOT NULL DEFAULT 0,
  s              int  NOT NULL DEFAULT 0,
  m              int  NOT NULL DEFAULT 0,
  l              int  NOT NULL DEFAULT 0,
  xl             int  NOT NULL DEFAULT 0,
  xxl            int  NOT NULL DEFAULT 0,
  xxxl           int  NOT NULL DEFAULT 0,
  total_qty      int  NOT NULL DEFAULT 0,
  delivery_date  date,
  delivery_notes text,
  status         text NOT NULL DEFAULT 'processing',
  order_type     text NOT NULL DEFAULT 'restock',
  created_at     timestamptz NOT NULL DEFAULT now()
);
ALTER TABLE orders ENABLE ROW LEVEL SECURITY;
CREATE POLICY "user read own orders"
  ON orders FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "user insert own orders"
  ON orders FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "auth read all orders"
  ON orders FOR SELECT USING (auth.role() = 'authenticated');
CREATE POLICY "auth update all orders"
  ON orders FOR UPDATE USING (auth.role() = 'authenticated');
